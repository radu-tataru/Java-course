Basic File Reading 

import java.io.*;
import java.nio.file.*;

public class FileOperations {
    
    // Reading entire file content
    public void readEntireFile(String filename) {
        try {
            String content = Files.readString(Paths.get(filename));
            System.out.println("File content: " + content);
        } catch (IOException e) {
            System.err.println("Error reading file: " + e.getMessage());
        }
    }
    
    // Reading file line by line
    public void readFileByLines(String filename) {
        try (BufferedReader reader = Files.newBufferedReader(Paths.get(filename))) {
            String line;
            int lineNumber = 1;
            while ((line = reader.readLine()) != null) {
                System.out.println(lineNumber + ": " + line);
                lineNumber++;
            }
        } catch (IOException e) {
            System.err.println("Error reading file: " + e.getMessage());
        }
    }
}

Writing to Files 

import java.io.*;
import java.nio.file.*;
import java.util.List;

public class FileWriting {
    
    // Writing string to file
    public void writeToFile(String filename, String content) {
        try {
            Files.writeString(Paths.get(filename), content);
            System.out.println("Content written to " + filename);
        } catch (IOException e) {
            System.err.println("Error writing file: " + e.getMessage());
        }
    }
    
    // Appending to file
    public void appendToFile(String filename, String content) {
        try {
            Files.writeString(Paths.get(filename), content, 
                StandardOpenOption.CREATE, StandardOpenOption.APPEND);
            System.out.println("Content appended to " + filename);
        } catch (IOException e) {
            System.err.println("Error appending to file: " + e.getMessage());
        }
    }
    
    // Writing multiple lines
    public void writeLines(String filename, List lines) {
        try {
            Files.write(Paths.get(filename), lines);
            System.out.println("Lines written to " + filename);
        } catch (IOException e) {
            System.err.println("Error writing lines: " + e.getMessage());
        }
    }
}

Working with Different File Formats 

import java.io.*;
import java.util.*;
import org.json.*;

public class FileFormats {
    
    // CSV file processing
    public void processCSV(String filename) {
        try (BufferedReader reader = Files.newBufferedReader(Paths.get(filename))) {
            String line;
            while ((line = reader.readLine()) != null) {
                String[] columns = line.split(",");
                System.out.println("Columns: " + Arrays.toString(columns));
            }
        } catch (IOException e) {
            System.err.println("Error processing CSV: " + e.getMessage());
        }
    }
    
    // JSON file processing
    public void processJSON(String filename) {
        try {
            String content = Files.readString(Paths.get(filename));
            JSONObject jsonObject = new JSONObject(content);
            
            // Access JSON data
            String name = jsonObject.getString("name");
            int age = jsonObject.getInt("age");
            
            System.out.println("Name: " + name + ", Age: " + age);
        } catch (IOException | JSONException e) {
            System.err.println("Error processing JSON: " + e.getMessage());
        }
    }
}

Path Operations 

import java.nio.file.*;

public class PathOperations {
    
    public void demonstratePathOperations() {
        // Creating paths
        Path currentDir = Paths.get(".");
        Path dataDir = Paths.get("data", "test-files");
        Path configFile = dataDir.resolve("config.properties");
        
        // Path information
        System.out.println("Current directory: " + currentDir.toAbsolutePath());
        System.out.println("Data directory: " + dataDir);
        System.out.println("Config file: " + configFile);
        
        // File existence and properties
        if (Files.exists(configFile)) {
            try {
                System.out.println("File size: " + Files.size(configFile) + " bytes");
                System.out.println("Is readable: " + Files.isReadable(configFile));
                System.out.println("Is writable: " + Files.isWritable(configFile));
            } catch (IOException e) {
                System.err.println("Error getting file properties: " + e.getMessage());
            }
        } else {
            System.out.println("Config file does not exist");
        }
        
        // Creating directories
        try {
            Files.createDirectories(dataDir);
            System.out.println("Directory created: " + dataDir);
        } catch (IOException e) {
            System.err.println("Error creating directory: " + e.getMessage());
        }
    }
}
