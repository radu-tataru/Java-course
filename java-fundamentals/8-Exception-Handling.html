Try-Catch-Finally Blocks 

public class ExceptionDemo {
    public void handleExceptions() {
        try {
            // Code that might throw an exception
            int result = divideNumbers(10, 0);
            System.out.println("Result: " + result);
        } catch (ArithmeticException e) {
            // Handle specific exception
            System.err.println("Cannot divide by zero: " + e.getMessage());
        } catch (Exception e) {
            // Handle any other exception
            System.err.println("Unexpected error: " + e.getMessage());
        } finally {
            // Always executes (cleanup code)
            System.out.println("Cleanup completed");
        }
    }
    
    private int divideNumbers(int a, int b) {
        return a / b;  // May throw ArithmeticException
    }
}

File Operations with Exception Handling 

import java.io.*;

public class FileExceptionDemo {
    public void readFileWithExceptions(String filename) {
        try (BufferedReader reader = new BufferedReader(new FileReader(filename))) {
            String line;
            while ((line = reader.readLine()) != null) {
                System.out.println(line);
            }
        } catch (FileNotFoundException e) {
            System.err.println("File not found: " + filename);
        } catch (IOException e) {
            System.err.println("Error reading file: " + e.getMessage());
        }
    }
}

Custom Exceptions 

// Custom exception class
class InvalidTestDataException extends Exception {
    public InvalidTestDataException(String message) {
        super(message);
    }
}

public class TestDataValidator {
    public void validateTestData(String data) throws InvalidTestDataException {
        if (data == null || data.trim().isEmpty()) {
            throw new InvalidTestDataException("Test data cannot be null or empty");
        }
        
        if (data.length() < 5) {
            throw new InvalidTestDataException("Test data must be at least 5 characters");
        }
        
        System.out.println("Test data is valid: " + data);
    }
    
    public void runValidation() {
        try {
            validateTestData("");
        } catch (InvalidTestDataException e) {
            System.err.println("Validation failed: " + e.getMessage());
        }
    }
}
