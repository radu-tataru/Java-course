Reading Stack Traces 

public class StackTraceDemo {
    public void demonstrateStackTrace() {
        try {
            methodA();
        } catch (Exception e) {
            // Print full stack trace
            e.printStackTrace();
            
            // Get specific information
            System.err.println("Error message: " + e.getMessage());
            System.err.println("Error type: " + e.getClass().getSimpleName());
        }
    }
    
    private void methodA() {
        methodB();
    }
    
    private void methodB() {
        methodC();
    }
    
    private void methodC() {
        // This will cause a NullPointerException
        String text = null;
        int length = text.length();
    }
}

Effective Logging 

import java.util.logging.*;

public class LoggingDemo {
    private static final Logger logger = Logger.getLogger(LoggingDemo.class.getName());
    
    public void demonstrateLogging() {
        // Different log levels
        logger.info("Application starting");
        logger.warning("This is a warning message");
        logger.severe("Critical error occurred");
        
        // Logging with variables
        String username = "testuser";
        logger.info("Processing user: " + username);
        
        // Logging exceptions
        try {
            riskyOperation();
        } catch (Exception e) {
            logger.log(Level.SEVERE, "Operation failed", e);
        }
    }
    
    private void riskyOperation() throws Exception {
        throw new Exception("Something went wrong");
    }
}

Common Debugging Scenarios 

public class DebuggingScenarios {
    
    // Null pointer debugging
    public void debugNullPointer() {
        String text = getText();
        
        // Always check for null before using
        if (text != null) {
            System.out.println("Length: " + text.length());
        } else {
            System.err.println("Text is null!");
        }
    }
    
    // Array bounds debugging
    public void debugArrayBounds() {
        int[] numbers = {1, 2, 3, 4, 5};
        int index = 10;
        
        // Check bounds before accessing
        if (index >= 0 && index < numbers.length) {
            System.out.println("Value: " + numbers[index]);
        } else {
            System.err.println("Index " + index + " is out of bounds");
        }
    }
    
    // Logic debugging with assertions
    public void debugLogic(int age) {
        // Use assertions to verify assumptions
        assert age >= 0 : "Age cannot be negative";
        
        if (age < 18) {
            System.out.println("Minor");
        } else if (age < 65) {
            System.out.println("Adult");
        } else {
            System.out.println("Senior");
        }
    }
    
    private String getText() {
        // This might return null in real scenarios
        return null;
    }
}
