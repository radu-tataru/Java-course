String Basics

public class StringDemo {
    public void stringOperations() {
        String text = "Hello World";
        
        // Length and character access
        int length = text.length();
        char firstChar = text.charAt(0);
        
        // Case operations
        String upper = text.toUpperCase();
        String lower = text.toLowerCase();
        
        // Substring operations
        String hello = text.substring(0, 5);
        String world = text.substring(6);
        
        // String comparison
        boolean isEqual = text.equals("Hello World");
        boolean isEqualIgnoreCase = text.equalsIgnoreCase("hello world");
        
        // Contains and search
        boolean contains = text.contains("World");
        int index = text.indexOf("World");
        
        // Split and join
        String[] words = text.split(" ");
        String joined = String.join("-", words);
        
        // Trim whitespace
        String trimmed = "  spaced  ".trim();
        
        System.out.println("Length: " + length);
        System.out.println("Uppercase: " + upper);
        System.out.println("Words: " + Arrays.toString(words));
    }
}

StringBuilder for Performance

public class StringBuilderDemo {
    public void efficientStringBuilding() {
        StringBuilder sb = new StringBuilder();
        
        // Append operations
        sb.append("Hello");
        sb.append(" ");
        sb.append("World");
        
        // Insert and delete
        sb.insert(5, " Beautiful");
        sb.delete(5, 15);
        
        // Convert to string
        String result = sb.toString();
        
        System.out.println("Result: " + result);
    }
}

Regular Expressions Basics

import java.util.regex.Pattern;

public class RegexDemo {
    public void validateStrings() {
        String email = "user@example.com";
        String phone = "123-456-7890";
        
        // Email validation pattern
        String emailPattern = "^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$";
        boolean isValidEmail = Pattern.matches(emailPattern, email);
        
        // Phone number validation
        String phonePattern = "\\d{3}-\\d{3}-\\d{4}";
        boolean isValidPhone = Pattern.matches(phonePattern, phone);
        
        System.out.println("Valid email: " + isValidEmail);
        System.out.println("Valid phone: " + isValidPhone);
    }
}
